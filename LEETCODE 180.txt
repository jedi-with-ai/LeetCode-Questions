SELECT DISTINCT(l1.num) AS ConsecutiveNums FROM Logs l1, Logs l2, Logs l3 
WHERE l1.num = l2.num 
AND  l2.num = l3.num
AND l2.id = l1.id - 1 
AND l3.id = l2.id - 1;


-- The idea is the value of the table needs to be equal to the other two rows but the id needs to be in a row that can be easily 
	done by id - 1 since they are auto increment and if there is hole in the id the above solution wont work but we have to have a system that 
	works when it says it auto-increases.
